function toggleConsoleFlag(name)
    console.flags[name] = not console.flags[name]
    console.addLine(name .. " = " .. tostring(console.flags[name]))
end

function parseFlagValue(value)
    local normalized = trim(tostring(value)):lower()
    if normalized == "1" or normalized == "true" or normalized == "on" then
        return true
    end
    if normalized == "0" or normalized == "false" or normalized == "off" then
        return false
    end
    return nil
end

local function setConsoleFlag(name, value)
    console.flags[name] = value
    console.addLine(name .. " = " .. tostring(console.flags[name]))
    if value and console.debugCommands[name] then
        console.debugCommands[name]()
    end
end

local function setOrToggleConsoleFlag(name, args)
    local value = parseFlagValue(args)
    if value == nil then
        toggleConsoleFlag(name)
        return
    end
    setConsoleFlag(name, value)
end

    end
end

local function executeGenericCommand(cmd, args)
    local handler = commands[cmd]
    if type(handler) == "function" then
        handler(args)
        return
    end
    if cmd:find("^debug_") then
        setOrToggleConsoleFlag(cmd, args)
        return
    end
    console.addLine("unknown command: " .. cmd .. ". type help")
end

local function executePlayerCommand(cmd, args)
    if cmd == "player_data_set" then
        local key, value = args:match("^(%S+)%s+(.+)$")
        if not key or value == "" then
            console.addLine("usage: player_data_set key value")
            return
        end
        console.playerData[key] = value
        console.addLine("player data set: " .. key .. "=" .. tostring(value))
        return
    end
    if cmd == "player_data_get" then
        local key = trim(args)
        if key == "" then
            console.addLine("usage: player_data_get key")
            return
        end
        console.addLine(key .. " = " .. formatValue(console.playerData[key]))
        return
    end
    if cmd == "player_data_show" then
        console.addLine("playerData = " .. formatValue(console.playerData))
        return
    end
    if cmd == "player_data_bukup" or cmd == "player_data_backup" then
        console.playerDataBackup = {}
        for k, v in pairs(console.playerData) do
            console.playerDataBackup[k] = v
        end
        console.addLine("playerData backup created")
        return
    end
    if cmd == "player_delete" then
        local key = trim(args)
        if key == "" then
            console.addLine("usage: player_delete key")
            return
        end
        console.playerData[key] = nil
        console.addLine("player data deleted: " .. key)
        return
    end
    if cmd == "player_verfy" or cmd == "player_verify" then
        local count = 0
        for _ in pairs(console.playerData) do
            count = count + 1
        end
        console.addLine("player data entries=" .. count .. ", backup=" .. tostring(next(console.playerDataBackup) ~= nil))
        return
    end
end
